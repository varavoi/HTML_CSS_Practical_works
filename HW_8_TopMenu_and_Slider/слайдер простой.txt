пишем в head кусок яваскрипт кода:
<script>
var slider = {
	slides:['ch01.jpg','ch02.jpg','ch03.jpg','ch04.jpg','ch05.jpg'],
	frame:0,
	set: function(image) {
		document.getElementById("scr").style.backgroundImage = "url("+image+")";
	},
	init: function() {
		this.set(this.slides[this.frame]);
	},
	left: function() { 
		this.frame--;
		if(this.frame < 0) this.frame = this.slides.length-1;
		this.set(this.slides[this.frame]);
	},
	right: function() { 
		this.frame++;
		if(this.frame == this.slides.length) this.frame = 0;
		this.set(this.slides[this.frame]);		
	}
};
window.onload = function() {
	slider.init();
	setInterval(function() {
		slider.right();
	},5000);
};
</script>



html:
<div id="main">
	<button class="left" onclick="slider.left();"><</button>
	<div id="scr"></div>
	<button class="right" onclick="slider.right();">></button>
</div>

.left {
	left:5px;
}
.right {
	right:5px; - смещение этих кнопок относительно слайдера по горизонтали












background-size - Масштабирует фоновое изображение согласно заданным размерам.

border-radius - устанавливает радиус скругления углов рамки. 

setInterval(function() { slider.right();},5000);}; - интервал для перелистывания картинок

#scr - область, где показываются слайды

#main  - "рамочка" для слайдера

slides:['ch01.jpg','ch02.jpg','ch03.jpg','ch04.jpg','ch05.jpg'] - имена файлов слайдов. должны храниться в той же папке, что и документ htm

button - создает на веб-странице кнопки

Событие onclick возникает при щелчке левой кнопкой мыши на элементе, к которому добавлен атрибут onclick.


background-color: transparent - фоновый цвет прозрачный


в button можно добавить:
border-radius: 100%; - скругление кнопки (100% - абсолютно круглая)
border: .2em solid rgba(255,255,255,1); - ширина и цвет контура кнопки
background-image:url(logo-finding-optimism_b.png); - фоновое изображение для кнопки

сам значок кнопки или фото кнопки пишется здесь:

<button class="left" onclick="slider.left();">пишется тег для вставки картинки</button>
